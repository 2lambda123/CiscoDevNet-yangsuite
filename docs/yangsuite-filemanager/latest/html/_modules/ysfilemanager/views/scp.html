import shutil
import tempfile

from django.views.decorators.http import require_POST
from django.contrib.auth.decorators import login_required
import paramiko
import scp
from yangsuite import get_logger, get_path

from .. import split_user_set, YSYangRepository

log = get_logger(__name__)

@require_POST
def scp_files_to_repo(request):
    """SCP files from the given remote directory to a repository.

    Returns:
      JsonResponse: {reply: result message}
    """
    repo = request.POST.get('repository')
    if not repo:
        return JsonResponse({}, status=400, reason="No repository specified")

    user = request.user.username
    owner, reponame = split_user_set(repo)

    if not reponame:
        return JsonResponse({}, status=400, reason="No repository name given")

    if user != owner and not request.user.is_superuser:
        return JsonResponse({}, status=403,
                            reason="Only superusers may modify the contents "
                            "of another user's repository")

    scratchdir = tempfile.mkdtemp(dir=get_path('scratch', user=user))
    try:
        with paramiko.SSHClient() as sshc:
            # Add new host keys, instead of rejecting as unknown
            sshc.set_missing_host_key_policy(paramiko.AutoAddPolicy())
            sshc.connect(request.POST.get('host'),
                         username=request.POST.get('ssh_username'),
                         password=request.POST.get('ssh_password'),
                         timeout=30)
            # Establish SCP session over the SSH session,
            # ensuring that wildcard '*' remains wild and unsanitized
            with scp.SCPClient(sshc.get_transport(),
                               sanitize=lambda x: x) as scpc:
                scpc.get(remote_path=os.path.join(
                    request.POST.get('remote_path'), "*.yang"),
                         local_path=scratchdir)

        targetrepo = YSYangRepository(owner, reponame)
        return JsonResponse({'reply': targetrepo.add_yang_files(scratchdir)})

    except paramiko.ssh_exception.AuthenticationException:
        return JsonResponse({}, status=401, reason="Authentication failed")

    except scp.SCPException as exc:
        return JsonResponse({}, status=500, reason=str(exc))

    finally:
        if os.path.exists(scratchdir):

            shutil.rmtree(scratchdir)

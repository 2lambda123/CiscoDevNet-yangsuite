[docs]def extra_node_data(stmt=None, parent_data=None, node=None):
    """Add tailf annotations to the node data.

    Callback function for :meth:`ysyangtree.ymodels.ParseYang.generate_tree`.

    Currently uses the following annotations:

    - `tailf:cli-sequence-commands` - specifies that the arguments in the model
      need to each be input in order to be accepted by the IOS XE CLI
    - `tailf:cli-break-sequence-commands` - negates a previous
      `cli-sequence-commands` annotation, meaning that this node and those
      after it no longer need to be specified in full order.
    - `tailf:cli-incomplete-command` - specifies that this leaf by itself
      does not constitute a complete CLI command.
      When combined with `cli-sequence-commands`, this means that the next leaf
      in the current container also needs to be included for the command
      to be considered valid.
    - `tailf:cli-optional-in-sequence` - specifies that the leaf may be omitted
      from the sequence of leafs under a `cli-sequence-commands` container.
      If not specified, this leaf must be specified if the next leaf in the
      container is also to be specified.

    Between the above annotations, we can determine any leafs that must precede
    a given leaf (due to `cli-sequence-commands` plus the absence of any
    `cli-optional-in-sequence` annotations, analogous to a YANG "when" clause)
    and any leafs that must follow a given leaf (due to the given leaf being
    `cli-incomplete-command`, analogous to a YANG "must" clause).

    For more information on these and other annotations, see:
    http://wwwin-people.cisco.com/phpetty/nso_man/man.5.tailf_yang_cli_extensions.html
    """  # noqa: E501

    # keep the below in alphabetical order so we don't duplicate by accident
    if stmt.search_one(('tailf-common', 'cli-break-sequence-commands')):
        node['data']['tailf:cli-break-sequence-commands'] = 'true'
    if stmt.search_one(('tailf-common', 'cli-incomplete-command')):
        node['data']['tailf:cli-incomplete-command'] = 'true'
    if stmt.search_one(('tailf-common', 'cli-optional-in-sequence')):
        node['data']['tailf:cli-optional-in-sequence'] = 'true'
    if stmt.search_one(('tailf-common', 'cli-sequence-commands')):
        node['data']['tailf:cli-sequence-commands'] = 'true'

    return node




[docs]def generate_replays(parseyang, base_xpath, category, output_path, **kwargs):
    """Generate a family of replays based on the given xpath root.

    Args:
      parseyang (ysyangtree.ymodels.ParseYang): Parsed YANG module tree
      base_xpath (str): Node 'xpath' to use as the root
        node to descend from.
      category (str): Subdirectory within output_path to group files into.
      output_path (str): Directory path to output files to.
        If not already ending with "/replays", that will be appended.
      kwargs: May include any of the following
        - generic_string_values: List of string values to use for
        create/merge and replace replays (in that order) for strings
        that don't have a defined pattern
    """
    global failed_xpaths
    failed_xpaths = {}

    if not output_path:
        test_dict = kwargs.get('test_dict', {})
        if not test_dict:
            log.error("No output directory")
            return {'errors': 'No output directory'}
        output_path = get_path('replays_dir', user=test_dict.get('user'))

    if output_path.endswith(os.sep):
        output_path = output_path[:-1]
    if not output_path.endswith("replays"):
        output_path = os.path.join(output_path, "replays")
    if not os.path.exists(output_path):
        os.makedirs(output_path)

    # Regenerate the parseyang tree with the additional attributes we need
    tree = parseyang.generate_tree(node_callback=extra_node_data)

    # Add parseyang for functions to use if needed
    kwargs['parser'] = parseyang

    BaseReplay.tw = parseyang.tw

    config_gen = ConfigGenerator(
        parseyang,
        base_xpath,
        category,
        output_path,
        **kwargs)

    if kwargs.get('gentype', 'all') == 'csv':
        return config_gen.make_tests()

    starting_point = walk_tree_to_xpath(tree, base_xpath, config_gen, **kwargs)
    initial_node = starting_point['node']
    initial_parent = starting_point['parent']
    if initial_node == tree:
        nodes = deque(
            NodeEntry(node, tree)
            for node in tree['children'])
        xpath_subst = []
    else:
        nodes = deque([
            NodeEntry(initial_node, initial_parent)
        ])
        xpath_subst = [(initial_node['data']['xpath_pfx'],
                        starting_point['xpath_after'])]

    while nodes:
        entry = nodes.popleft()
        node = entry.node
        parent = entry.parent
        data = node['data']
        nodetype = data['nodetype']
        xpath_pfx = data['xpath_pfx']
        # Make substitutions in xpath as appropriate
        xpath_pfx = xpath_substitution(xpath_pfx, xpath_subst)
        access = data.get('access', '').strip()
        deviate = data.get('deviation', None)
        status = data.get('status', None)
        if deviate == 'not-supported' or status == 'obsolete':
            continue

        log.info("Constructing replay parameters for xpath:\n%s", xpath_pfx)

        if nodetype == 'list':
            config_gen.groups.append(
                ListReplay(node, parent, xpath_pfx, xpath_subst, **kwargs)
            )

        if nodetype == 'container':
            config_gen.groups.append(
                ContainerReplay(node, parent, xpath_pfx, xpath_subst, **kwargs)
            )

        children = xpath_children(node)

        # IOS XE structures interface selection as a choice-case structure
        # where each interface type is a case. We don't need to generate
        # replays for each and every possible type of interface;
        # furthermore, the specific interface type(s) available on a device
        # will vary by device. Just pick a single type:
        if any("AppNav-Compress" in child['text'] for child in children):
            children = [child for child in children
                        if (child['text'] == 'AppNav-Compress' or
                            ':AppNav-Compress' in child['text'])]

        nodes.extendleft(NodeEntry(child, node)
                         for child in reversed(children))

        # Replace the specific IOS XE type with a variable to be
        # supplied by the user
        if 'AppNav-Compress' in xpath_pfx:
            xpath_pfx = re.sub(r'AppNav-Compress', '_-interface_type-_',
                               xpath_pfx)

        if nodetype == 'leaf-list':
            if access in ['read-only', 'write']:
                continue
            config_gen.groups.append(
                LeafListReplay(node, parent, xpath_pfx, xpath_subst, **kwargs)
            )

        if nodetype == 'leaf':
            if access == 'write':
                continue
            config_gen.groups.append(
                LeafReplay(node, parent, xpath_pfx, xpath_subst, **kwargs)
            )

    replay_generation_status = config_gen.make_tests()
    replay_generation_status['failed_xpaths'] = failed_xpaths

    config_gen.reset()

    return replay_generation_status




[docs]def add_extra_nodes(extra_nodes, more_extra_nodes):
    """Add extra nodes like mandatory children or list keys.

    Args:
      extra_nodes (list): List of xpath/value dicts needed for replay.
      more_extra_nodes (list): Additional xpath/value dicts for replay.
    """
    if not more_extra_nodes:
        return
    if not extra_nodes:
        extra_nodes += more_extra_nodes
        return

    for extra in extra_nodes:
        for new_extra in more_extra_nodes:
            if re.sub(RE_FIND_KEYS, '', extra['xpath']) == \
                    re.sub(RE_FIND_KEYS, '', new_extra['xpath']):
                more_extra_nodes.remove(new_extra)
                break
            if extra['xpath'] == new_extra['xpath']:
                more_extra_nodes.remove(new_extra)
                break
    extra_nodes += more_extra_nodes

[docs]def xpath_substitution(xpath, xpath_subst):
    """Fill in list keys, etc. in the given XPath as appropriate.

    Args:
      xpath (str): Base XPath to perform substitutions against
      xpath_subst (list): List of (before, after) string substitutions

    Returns:
      str: Substituted XPath.
    """
    for base, subst in xpath_subst:
        # The additional substring checks are to make sure we don't, say,
        # use a subst meant for "/foo/ip" when we have "/foo/ipv6"
        if xpath.startswith(base) and (
                xpath[len(base):] == '' or
                xpath[len(base):].startswith('/')
        ):
            xpath = subst + xpath[len(base):]

    return xpath




[docs]def walk_tree_to_xpath(tree, target_xpath, config_gen, **kwargs):
    """Walk the given node data tree until reaching the desired node.

    Args:
      tree (dict): YANG model tree to walk
      target_xpath (str): XPath of desired target node
      kwargs: Takes in any available kwargs from generate_replays,
        check docstring for possible values.

    Returns:
      dict: with the following keys.
        - 'node' (node dict with data and children)
        - 'parent' (node dict with data and children)
        - 'xpath' (xpath updated with list keys as appropriate)
        - 'mandatory_nodes' (list of (node, xpath) tuples)
    """
    implicit_prefix = tree['data']['prefix']
    target_xpath = xpath_to_xpath_pfx(target_xpath, implicit_prefix)
    xpath_so_far = ''
    changes_so_far = ''
    xpath_subst = []
    current_node = tree
    parent = current_node

    for token, keys_values, kv_tokens, _ in xpath_iterator(target_xpath):
        xpath_so_far += '/' + token
        changes_so_far += '/' + token

        prefix, localname = token.split(':', 1)

        # Find the next node we'll descend to
        for child in xpath_children(current_node):
            if child['text'] == token or child['text'] == localname:
                next_node = child
                parent = current_node
                break
        else:
            raise KeyError('Child "{0}" not found at {1} in {2}'
                           .format(token, xpath_so_far, target_xpath))

        xpath_so_far += ''.join(kv_tokens)
        if next_node['data']['nodetype'] == 'list':
            if xpath_so_far != target_xpath:
                listreplay = ListReplay(
                    next_node, parent, changes_so_far, xpath_subst, **kwargs
                )
                # Don't generate replays until we get to target
                listreplay.pre_target = True
                if kwargs.get('key_leafref_setting') != 'set-leafref':
                    # Move the referred-to leaf(s) into pre_nodes
                    for xpath, nde in listreplay.referred_to_nodes:
                        if nde not in listreplay.pre_nodes:
                            listreplay.pre_nodes.append(nde)
                config_gen.groups.append(listreplay)
                changes_so_far = listreplay.new_xpath

        if current_node != tree and \
                current_node['data']['nodetype'] in ['list', 'container']:
            mandatory_children_of_node(
                current_node,
                xpath_subst,
                BaseReplay.mandatory_nodes,
                **kwargs
            )

        parent = current_node
        current_node = next_node

        if current_node == tree or xpath_so_far == target_xpath:
            # On target so remove all "replace" values from mandatory nodes.
            for node in BaseReplay.mandatory_nodes:
                for cfg in node.get('configs', []):
                    cfg.pop('replace_value', '')
            break

    return {
        'node': current_node,
        'parent': parent,
        'xpath_after': xpath_substitution(changes_so_far, xpath_subst)
    }




[docs]def xpath_children(node):
    """Get the next children in the xpath.

    I.e., skip through choice/case nodes as they aren't part of the xpath.

    Args:
      node (dict)

    Returns:
      list: of node dicts.
    """
    children = []
    for child in node.get('children', []):
        # Non schema nodes don't have xpath_pfx
        # This will prevent xpath_children from throwing an Error 500
        if child['data']['nodetype'] in NON_SCHEMA_NODETYPES:
            continue
        elif child['data']['nodetype'] in NON_DATA_NODETYPES:
            children += xpath_children(child)
        else:
            children.append(child)

    return children




[docs]def mandatory_children_of_node(root_node, xpath_subst,
                               mandatory_nodes=[], **kwargs):
    """Get the descendants of the given node which are themselves mandatory.

    At present, handles any ``mandatory`` leaf and choice nodes.

    - TODO: handle list/leaf-list with ``min-elements``.

    Args:
      node (dict): Representing a node in a yangtree
      xpath_subst (list): Of (before, after) XPath string substitutions

    Returns:
      list: of dicts with keys 'xpath', 'value'.
    """
    root_data = root_node['data']
    nodetype = root_data['nodetype']
    if nodetype != 'list' and nodetype != 'container':
        return
    for child in root_node.get('children', []):
        child_data = child['data']
        if child_data.get('deviation', '') == 'not-supported':
            continue
        if child_data.get('status', '') == 'obsolete':
            continue

        if child_data['nodetype'] == 'container':
            mandatory_children_of_node(
                child, xpath_subst, mandatory_nodes, **kwargs
            )
        elif child_data['nodetype'] != 'choice':
            if child_data.get('mandatory', '') != 'true':
                continue
            child_value = values_for_leaf_data(
                child,
                root_node,
                **kwargs
            )
            if not child_data:
                continue
            new_node = {
                'xpath': xpath_substitution(child_data['xpath_pfx'],
                                            xpath_subst),
                'value': child_value[0],
                'replace_value': child_value[1],
                'datatype': child_data.get(
                    'basetype',
                    child_data.get('datatype', '')
                ),
                'default': child_data.get('default', ''),
                'nodetype': child_data.get('nodetype', '')
            }
            if not hasattr(child, 'configs'):
                child['configs'] = [new_node]
            else:
                child['configs'].append(new_node)
            for n in mandatory_nodes:
                if n['data']['xpath'] == child_data['xpath']:
                    break
            else:
                mandatory_nodes.append(child)
        elif child_data['nodetype'] == 'choice':
            ChoiceReplay(child, root_node, root_node['data']['xpath_pfx'],
                         xpath_subst, **kwargs)
        else:
            log.warning("Unhandled mandatory node at %s",
                        child_data['xpath'])




[docs]def values_for_leaf_data(node, parent, generic_string_values={}, **kwargs):
    """Given a yangtree 'data' dict, determine candidate leaf values.

    Args:
      node (dict): As in yangtree node, the leaf or leaf-list node.
      parent (dict): The leaf's or leaf-list's immediate parent node.
      generic_string_values (list): A list of string values to use for replays
        that the user can provide. Currently the first value corresponds to a
        value for create/merge replays and the second to a value for replace
        replays.
      kwargs: Takes in any preconfigured values

    Returns:
      list: of zero or more (usually, but not always, 2) candidate
      values for this leaf.
    """
    data = node['data']
    nodename = data.get('name', '')
    basetype = data.get('basetype')
    datatype = data.get('datatype')
    maximum = data.get('max')
    minimum = data.get('min')
    options = list(data.get('options', []))
    patterns = list(data.get('patterns', []))
    xpath = data.get('xpath', '')
    leafref_path = data.get('leafref_path', '')
    default = data.get('default')

    # Often, model developers don't specify an exact range for int values,
    # and use a type such as uint[8|16|32|64] that has a default minimum of 0,
    # but the actual data implementation doesn't actually permit zero.
    # Similarly, the actual maximum may be much less than 2^n - 1.
    # For now, while having basic model support work at all is the goal,
    # we will protect the user accordingly:
    if minimum == '0' and maximum in ['255', '65535', '4294967295',
                                      '18446744073709551615']:
        minimum = '10'
        maximum = '20'

    values = []

    # Interface leaves are likely to be different from target device to device,
    # so let's use a variable for those. The following are heuristics to try
    # to guess that a leaf is an interface.
    if (
            # openconfig patterns
            re.search("/([^:]+:)?interfaces?/([^:]+:)?(name|id)", xpath)
            or
            re.search("/([^:]+:)?interfaces?/[^/]+/([^:]+:)?(name|id)", xpath)
            or
            re.match("/([^:]+:)?interfaces/([^:]+:)?interface/([^:]+:)?name",
                     leafref_path)
            or
            # IOS XR native
            datatype == 'xr:Interface-name'
            or
            # NX-OS native types
            datatype in ['nw_IfId', 'acl_IfName', 'ipqos_IfName', 'ac_PortId']
    ):
        values += ['_-interface_name_1-_', '_-interface_name_2-_']
    elif 'AppNav-Compress' in nodename:
        # IOS XE native, see parallel logic in generate_replays
        values += ['_-interface_id_1-_', '_-interface_id_2-_']
    elif minimum and maximum:
        if minimum == maximum:
            values += [minimum]
        else:
            # We're not focused on boundary testing at the moment,
            # so rather than using [minimum, maximum], let's use values
            # from the middle of the range that are most likely "safe".
            mini = int(float(minimum))
            maxi = int(float(maximum))
            values += [str(mini + int((maxi - mini) / 2)),
                       str(mini + int((maxi - mini) / 2) + 1)]
    elif options:
        # For interfaces, special-case to ethernet as the most likely
        if 'ianaift:ethernetCsmacd' in options:
            values += ['ianaift:ethernetCsmacd']
        else:
            # Remove enum values that are not valid config targets:
            for exclusion in ['invalid', 'undefined', 'unspecified',
                              'unknown', 'none', 'failed', 'illegal']:
                if exclusion in options and len(options) > 1:
                    options.remove(exclusion)
            # Also avoid selecting the default value as it may behave a bit
            # differently from non-default values (e.g., on NX-OS, where the
            # default value is implicitly configured)
            if default in options and len(options) > 1:
                options.remove(default)

            # Even after the above filter, it's possible that the first
            # and last enum values are special in some way, so to reduce our
            # risk slightly, pick from the middle of the list if possible
            if len(options) <= 2:
                values += options
            else:
                values += [options[int(len(options) / 2)],
                           options[int(len(options) / 2) + 1]]
    elif datatype == 'empty':
        values += ['']
    elif datatype == 'leafref' or basetype == 'leafref':
        # Native models (XR, XE, NX) do not generally use leafrefs.
        # These are mostly seen in OpenConfig and IETF models.
        # In OpenConfig, the most common pattern is a list key "foo" which
        # is a leafref to either "../config/foo" or "../state/foo".
        # Since we have the parent, we can easily walk to the referenced leaf.
        if re.match('^../[^/]*(config|state)/', leafref_path):
            if len(data['leafref_path'].split('/')) == 3:
                _, sibling_name, target_name = data['leafref_path'].split('/')
                for sibling in parent['children']:
                    if (sibling['text'] == sibling_name or
                            sibling['data']['name'] == sibling_name):
                        for nephew in sibling['children']:
                            if (nephew['text'] == target_name or
                                    nephew['data']['name'] == target_name):
                                values += values_for_leaf_data(
                                    nephew,
                                    sibling,
                                    **kwargs
                                )
                                break
                        else:
                            # Just in case we didn't find the expected leaf
                            values += ['FIXME', 'FIXMETOO']
                        break
                else:
                    """Just in case we didn't find the
                       expected config/state branch"""
                    values += ['FIXME', 'FIXMETOO']
        else:
            # More complex XPaths, for example:
            # "../../index"
            # "../../../../../../../rt:ribs/rt:rib/rt:name"
            # "/vlan:vlans/vlan:vlan/vlan:config/vlan:vlan-id"
            # At present we don't attempt to handle these as such.
            values += ['FIXME', 'FIXMETOO']

    elif 'ipv4-address' in datatype or 'address_Ipv4' in datatype:
        # The inet:ipv4-address typedef pattern uses some advanced regex
        # features not supported by xeger.
        # The address_Ipv4 datatype in NX-OS uses the same regex pattern.
        # So provide values ourselves:
        if nodename and 'mask' in nodename:
            values += ['255.0.0.0', '255.255.255.0']
        else:
            values += ['10.1.1.1', '192.168.1.1']
    elif 'ipv4-host' in datatype:
        values += ['10.1.1.1', 'host.example.com']
    elif 'ipv4-prefix' in datatype:
        values += ['10.0.0.0/8', '192.168.1.0/24']
    # The patterns used for inet:ipv6-* and ios-types:ipv6-*
    # are underspecified and allow garbage values that are not actually
    # valid IPv6 addresses.
    # The address_Ipv6 datatype in NX-OS uses the same regex pattern.
    # So let's provide sane values ourselves:
    elif 'ipv6-address' in datatype or 'address_Ipv6' in datatype:
        values += ['2000::1', '2001::2']
    elif 'ipv6-prefix' in datatype:
        values += ['2001::/64', 'C15C:0D06:F00D::/64']
    elif 'ip-address' in datatype:
        values += ['10.1.1.1', '2001::2']
    elif 'inet:host' in datatype or 'ip-host' in datatype:
        values += ['10.1.1.1', '2001::2', 'host.example.com']
    elif 'address_Mac' in datatype:
        # NX-OS MAC address type is under-specified and allows values
        # with mixed syntax like 01:02-03:04-05-06 that aren't actually valid.
        # So provide values ourselves:
        values += ['C1:5C:00:00:00:01', '10:00:00:C1:5C:00']
    elif datatype == 'yang:date-and-time':
        values += [datetime.now().isoformat(), datetime.now().isoformat()]

    elif patterns:
        for pattern in patterns:
            try:
                # try a reverse regex to find a string
                values += safe_xegers(pattern)
            except Exception:
                continue

        if not values:
            # Give default strings as backup hoping they may pass some tests
            values = ['patternstring', 'replacestring']
            log.warning(
                "Reverse regex for patterns not found\n{0}\n\n{1}".format(
                    str(patterns),
                    'Substituting these strings ' + str(values)
                )
            )

    elif datatype == 'string' or basetype == 'string':
        # Note that a generic string *may* include all sorts of special
        # characters, including \t, \r, \n, etc. However, in many of
        # our YANG models, a string is expected to not contain whitespace
        # or other special characters. Ideally these models would provide
        # "pattern" restrictions specifying what they *really* expect,
        # but in practice that isn't feasible.
        create_merge_val = generic_string_values.get('create-merge')
        replace_val = generic_string_values.get('replace')
        if not create_merge_val:
            create_merge_val = 'genericstring'
        if not replace_val:
            replace_val = 'arbitrarystring'

        generic_string_values_list = [create_merge_val, replace_val]

        values += generic_string_values_list
    elif datatype == 'union' or basetype == 'union' or 'members' in data:
        members = data.get('members', [])
        for member in members:
            member['xpath_pfx'] = data.get('xpath_pfx', '/')
            values += values_for_leaf_data(
                {'data': member},
                parent,

        """Generate an appropriate name for a constructed replay.

        Note that yangsuite-testmanager identifies generated replays based on
        the naming pattern, so any changes to this need to be validated
        against that!

        Args:
          counter (int): Numerical ID associated with this replay
          prefix (str): XPath this replay applies to
          suffix (str): Descriptive string for this replay

        Returns:
          str: Of no more than about 72 characters in length, if possible.

        Examples:

        # Simple case - no trimming required
        1, "/native/banner/exec/banner", "basic merge"
        # Returns
        '00001 - /native/banner/exec/banner - basic merge'

        # XPath needs namespace prefixes stripped and to be trimmed as well
        150, '<long path>/pfx:redundancy/pfx:minimum-preferred-links', 'extra'
        # Returns
        '00150 - .../redundancy/minimum-preferred-links - extra'

        # Also requires stripping out of list keys
        1, '/interface[active="act"]/physical/slot', "basic replace"
        # Returns
        '00001 - .../physical/slot - basic replace'

        # Both prefix and suffix need to be trimmed for length
        """
        # Strip out the namespace prefixes and keys from the xpath
        prefix = self._clean_xpath(prefix)

        # Trim the suffix until it's not too long
        while len(suffix) > 20 and ' ' in suffix:
            suffix = '...' + ' '.join(suffix.split(' ')[1:])

        # Trim the xpath until it's not too long
        while len(prefix) > 40 and '/' in prefix[4:]:
            # print("{0:02} : {1}".format(len(prefix), prefix))
            prefix = '.../' + '/'.join(prefix.split('/')[2:])

        return "{0:05} - {1} - {2}".format(counter, prefix, suffix)

        """Find nearest upstream list or precense container of mandatory node.

        Leaf nodes with mandatory "true" are required for all configurations
        that fall within a non-presence container.
        """
        tw = self.parseyang.tw
        BaseReplay.mandatory_nodes = sorted(
            BaseReplay.mandatory_nodes,
            key=lambda mnd: mnd['id'])
        for node in BaseReplay.mandatory_nodes:
            xp = node.get('data').get('xpath')
            par_xp = xp[:xp.rfind('/')]
            while par_xp:
                # Move up the xpath until you find a list or presence container
                pnode = tw.get_node_by_xpath(par_xp)
                if pnode:
                    data = pnode.get('data', {})
                    if data.get('nodetype') == 'list' or data.get('presence'):
                        if par_xp in self.mandatory_cutoffs:
                            self.mandatory_cutoffs[par_xp].append(node)
                        else:
                            self.mandatory_cutoffs[par_xp] = [node]
                        break
                par_xp = par_xp[:par_xp.rfind('/')]
            else:
                if '/' in self.mandatory_cutoffs:
                    self.mandatory_cutoffs['/'].append(node)
                else:
                    self.mandatory_cutoffs['/'] = [node]

        """ Function to add mandatory nodes to configs."""
        tw = self.parseyang.tw
        grp_xpath = group.data.get('xpath')
        grp_schema = group.data.get('schema_node_id')
        choice_mandatory = []
        found = False
        for node in cutoff_node:
            schemapath = node['data'].get('schema_node_id')
            node_xpath = node['data'].get('xpath')
            # To check if mandatory node is under choice path
            if schemapath != node_xpath:
                choice_mandatory.append(node)
            else:
                self.add_mandatory(node, group)

        for cnode in choice_mandatory:
            cxpath = cnode['data'].get('xpath')
            # If the current node is same as mandatory node
            if cxpath == grp_xpath:
                found = True
                continue
            # To check if current node and
            # mandatory node are siblings
            par_xp = grp_xpath[:grp_xpath.rfind('/')]
            cxpath = cxpath[:cxpath.rfind('/')]
            if par_xp == cxpath:
                found = True
                # To check if current node is under the
                # mandatory choice node and if true just skip adding
                # the extra mandatory node as current node will be one
                # of the case statement.
                if group.node.get('mandatory_choice', False):
                    continue
                self.add_mandatory(cnode, group)
            else:
                # To check if mandatory nodes reside under presence
                # container.
                pnode = tw.get_node_by_xpath(cxpath)
                if pnode:
                    if cxpath in grp_xpath:
                        if pnode['data'].get('presence'):
                            boundary = grp_xpath[len(cxpath):]
                            if not boundary or \
                               boundary .startswith('/'):
                                found = True
                                self.add_mandatory(cnode, group)
                        # To check the current node and mandatory node
                        # are under same case statements, and add the
                        # mandatory nodes accordingly.
                        elif pnode['children']:
                            for ch in pnode['children']:
                                if ch.get('data', {}).get('nodetype') == \
                                   'choice':
                                    for case in self.get_case_nodes(ch):
                                        case_path = case.get('data').get(
                                            'schema_node_id')
                                        if case_path in grp_schema:
                                            boundary = grp_schema[len(
                                                case_path):]
                                            if not boundary or \
                                               boundary.startswith('/'):
                                                found = True
                                                self.add_mandatory(
                                                    cnode, group)
        # If any of the above conditions doesn't satisfy
        # and if we still have a choice/case mandatory node
        # just add the nodes from the first case.
        if not found and len(choice_mandatory) > 0:
            cnode = choice_mandatory[0]
            cpath = cnode['data'].get('xpath')
            self.add_mandatory(cnode, group)
            for node in choice_mandatory:
                if cnode == node:
                    continue
                nodepath = node['data'].get('xpath')
                if nodepath[:nodepath.rfind('/')] == \
                   cpath[:cpath.rfind('/')]:
                    self.add_mandatory(node, group)

    def xpath(self, xpath):
        if not xpath:
            log.error('No leafref path for {0}'.format(self.name))
            self._config = {}
            return
        leafref_tokens = xpath.split("/")
        leafref_path = ""
        # Parse leafref path to find data in node
        for lrt in leafref_tokens:
            if lrt.count(':') == 1:
                lrt = lrt.split(':')[1]
            if leafref_path:
                leafref_path += "/"
            leafref_path += lrt
        if leafref_tokens[0] == "":
            # Absolute path reference
            if not (leafref_path.startswith("/")):
                leafref_path = "/" + leafref_path
        elif leafref_path.startswith("../"):
            # Relative path reference
            base_xpath = self.base_data['xpath']
            ref_base_xpath = "/".join(base_xpath.split("/")[:-1])
            leafref_path = leafref_path[3:]
            while leafref_path.startswith("../"):
                leafref_path = leafref_path[3:]
                ref_base_xpath = "/".join(ref_base_xpath.split("/")[:-1])
            if ref_base_xpath == "/".join(base_xpath.split("/")[:-1]) and \
                    self.base_data.get('key') == 'true':
                # referred-to IS IN THE LIST ENTRY OF REFERRING KEY!
                leafref_path = ref_base_xpath + "/" + leafref_path
            else:
                leafref_path = base_xpath + "/" + leafref_path
        parser = self.kwargs.get('parser')
        node = parser.tw.get_node_by_xpath(leafref_path, no_prefix=True)
        # TODO: leafref in leafref maybe?
        if not node:
            log.warning(
                'Could not find referred-to leaf: {0}'.format(leafref_path)
            )
            # Put something in there.
            self._xpath = leafref_path
            # Pull values from base node
            self.get_values(self.node, self.parent, **self.kwargs)
            self._config = {
                'xpath': self._xpath,
                'value': self.value,
                'replace_value': self.replace_value,
                'datatype': self.datatype,
                'default': self.default,
                'nodetype': self.node['data']['nodetype']
            }
        else:
            self._xpath = xpath_substitution(
                node['data'].get('xpath_pfx'),
                self.xpath_subst
            )
            self.config = node
